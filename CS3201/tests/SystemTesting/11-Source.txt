procedure p1{
	if sun then { 							\\1
		man = 15 - 3 + 20 * 2;					\\2
		man = 3;						\\3
	} else {
		if master then {					\\4 	
			while doctor { 					\\5
				doctor = 8 + 8;				\\6
			}
			ghost = doctor + 1 * 7;				\\7	
			heir = 5 * 11 * doctor - sun * 9;		\\8
			good = running + good - running;		\\9
			man = running - master - 10 * sun - sun;	\\10
			good = 12 + 14 - ghost - man;			\\11
			if psy then { 					\\12
				psy = 20;				\\13
			} else {
				good = 18 * good * heir;		\\14	
			}
			if running then {				\\15 	
				good = 3 + good * heir + psy;		\\16
			} else {
				cho = master + 8 * 10 + 5;		\\17	
				call p3;				\\18
				sun = good - doctor;			\\19
				ghost = 3 * cho + good * 13;		\\20
			}
			good = 19 * 8 * heir;				\\21	
			sun = heir;					\\22
		} else {
			psy = sun;					\\23	
			if ghost then { 				\\24
				ghost = 3 - heir + 14 * 15;		\\25
			} else {
				sun = good - 16 + 17 - 1 * psy;		\\26	
			}
			call p2;					\\27	
			sun = 3 * 15;					\\28
		}
	}
	man = 3 - psy + 17 - 20 * master;				\\29	
	doctor = 17 + good * cho * 14 - doctor;				\\30
}
procedure p2{

			} else {
				sun = 18 + 18 - ghost + 10;		\\39	
			}
		}
		cho = psy - doctor;					\\40	
		sun = 20 + 9 - 6 * 2;					\\41

procedure p4{
	man = psy;							\\114
	while running { 						\\115
		while psy { 						\\116
			man = 1;					\\117
			psy = 1 * 17 + master;				\\118
		}
		while doctor { 						\\119	
			while ghost { 					\\120
				heir = ghost - psy - 4 - 13;		\\121
			}
		}
		if psy then { 						\\122
			good = psy;					\\123
			ghost = 7;					\\124
		} else {
			if man then { 					\\125
				heir = psy - 13 + doctor * 3 - 16;	\\126
			} else {
				call man;				\\127
				master = 20 - master;			\\128
				man = cho;				\\129
			}
		}
	}
	while heir { 							\\130	
		ghost = 13 - 12 + psy + 20;				\\131
	}
	while good { 							\\132
		doctor = ghost + sun * doctor - ghost + 11;		\\133
	}
	ghost = running - 6 * 6;					\\134
	good = 18;	
	while master { 	
		if master then { 	
			while master { 	
				heir = 3 - doctor - 17 + ghost;	
			}
			doctor = 14 - 5 + 11;	
			while cho { 	
				heir = ghost - master;	
				psy = 19 - 5 - psy * sun + 19;	
				running = 11 + 13 + 7;	
			}
			heir = heir * running * 9 - doctor;	
		} else {
			master = 2 + heir - 14 - 19 * sun;	
			while master {
				cho = good * cho;	
			}
		}
		if heir then { 	
			master = psy + ghost * 4 * 13;	
		} else {
			if heir then { 
				heir = ghost + master - 12;
				man = psy + 9 * 20 + 6;	
			} else {
				man = 17 * 5 + doctor;	
				cho = heir + sun * psy + 10 + good;
			}
		}
		while man { 	
			if running then { 	
				running = doctor - ghost * psy + doctor;	
			} else {
				doctor = doctor + running - 20 * cho;
				master = 1;
			}
			master = man + 1 * ghost + 1 * 4;	
		}
		master = man * ghost;	
		while good { 	
			cho = 4 - 7 - sun * running + good;	
		}
	}
}

procedure man {

	while sun {

			good = 9 + 5 + 15 * 2;	
			running = 1 * man + 17 - doctor;	
		} else {
			good = heir + 20 + 4 + sun;
		}
	}
}
